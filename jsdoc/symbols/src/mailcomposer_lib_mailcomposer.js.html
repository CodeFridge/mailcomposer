 80 /**
 81  * <p>Normalizes a key name by capitalizing first chars of words</p>
 82  * 
 83  * <p><code>x-mailer</code> will become <code>X-Mailer</code></p>
 84  * 
 85  * <p>Needed to avoid duplicate header keys</p>
 86  * 
 87  * @param {String} key Key name
 88  * @return {String} First chars uppercased
 89  */
 90 MailComposer.prototype.normalizeKey = function(key){
 91     return (key || "").toString().trim().
 92         toLowerCase().
 93         replace(/^\S|[\-\s]\S/g, function(c){
 94             return c.toUpperCase();
 95         }).replace(/^MIME\-/i, "MIME\-");
