 27  * @constructor
 28  * @param {Object} [options] Optional options object
 29  */

 31     Stream.call(this);
 32     
 33     this.options = options ||Â {};
 34

 77 /**
 78  * <p>Resets and initializes MailComposer</p>
 79  * 
 80  * <p>Setting an option overwrites an earlier setup for the same keys</p>
 81  * 
 82  * <p>Possible options:</p>
 83  * 
 84  * <ul>
 85  *     <li><b>from</b> - The e-mail address of the sender. All e-mail addresses can be plain <code>sender@server.com<code> or formatted <code>Sender Name <sender@server.com></code></li>
 86  *     <li><b>to</b> - Comma separated list of recipients e-mail addresses that will appear on the <code>To:</code> field</li>
 87  *     <li><b>cc</b> - Comma separated list of recipients e-mail addresses that will appear on the <code>Cc:</code> field</li>
 88  *     <li><b>bcc</b> - Comma separated list of recipients e-mail addresses that will appear on the <code>Bcc:</code> field</li>
 89  *     <li><b>replyTo</b> - An e-mail address that will appear on the <code>Reply-To:</code> field</li>
 90  *     <li><b>subject</b> - The subject of the e-mail</li>
 91  *     <li><b>body</b> - The plaintext version of the message</li>
 92  *     <li><b>html</b> - The HTML version of the message</li>
 93  * </ul>
 94  * 
 95  * @param {Object} options Message related options
 96  */
 97 MailComposer.prototype.setMessageOption = function(options){
 98     var fields = ["from", "to", "cc", "bcc", "replyTo", "subject", "body", "html"],
 99         rewrite = {"sender":"from", "reply_to":"replyTo"};
100     
101     options = options || {};
102     
103     var keys = Object.keys(options), key, value;
104     for(var i=0, len=keys.length; i<len; i++){
105         key = keys[i];
106         value = options[key];
107         
108         if(key in rewrite){
109             key = rewrite[key];
110         }
111         
112         if(fields.indexOf(key) >= 0){
113             this._message[key] = this.handleValue(key, value);
114         }
115     }
116 };

118 /**
119  * <p>Handles a message object value, converts addresses etc.</p>
120  * 
121  * @param {String} key Message options key
122  * @param {String} value Message options value
123  * @return {String} converted value
124  */
125 MailComposer.prototype.handleValue = function(key, value){
126     key = (key || "").toString();
127     
128     var addresses;
129     
130     switch(key){
131         case "from":
132         case "to":
133         case "cc":
134         case "bcc":
135         case "replyTo":
136             value = (value ||Â "").toString().replace(/\r?\n|\r/g, " ");
137             addresses = mimelib.parseAddresses(value);
138             this._envelope[key] = addresses.map((function(address){
139                 if(this.hasUTFChars(address.address)){
140                     return toPunycode(address.address);
141                 }else{
142                     return address.address;
143                 }
144             }).bind(this));
145             return this.convertAddresses(addresses);
146         case "subject":
147             value = (value ||Â "").toString().replace(/\r?\n|\r/g, " ");
148             return this.encodeMimeWord(value, "Q", 78);
149     }
150     
151     return value;
152 };

154 /**
155  * <p>Handles a list of parsed e-mail addresses, checks encoding etc.</p>
156  * 
157  * @param {Array} value A list or single e-mail address <code>{address:'...', name:'...'}</code>
158  * @return {String} Comma separated and encoded list of addresses
159  */
160 MailComposer.prototype.convertAddresses = function(addresses){
161     var values = [], address;
162     
163     for(var i=0, len=addresses.length; i<len; i++){
164         address = addresses[i];
165         
166         if(address.address){
167         
168             // if user part of the address contains foreign symbols
169             // make a mime word of it
170             address.address = address.address.replace(/^.*?(?=\@)/, (function(user){
171                 if(this.hasUTFChars(user)){
172                     return mimelib.encodeMimeWord(user, "Q");
173                 }else{
174                     return user;
175                 }
176             }).bind(this));
177             
178             // If there's still foreign symbols, then punycode convert it
179             if(this.hasUTFChars(address.address)){
180                 address.address = toPunycode(address.address);
181             }
182         
183             if(!address.name){
184                 values.push(address.address);
185             }else if(address.name){
186                 if(this.hasUTFChars(address.name)){
187                     address.name = this.encodeMimeWord(address.name, "Q", 78);
188                 }else{
189                     address.name = address.name;
190                 }
191                 values.push('"' + address.name+'" <'+address.address+'>');
192             }
193         }
194     }
195     return values.join(", ");
196 };

198 /**
199  * <p>Adds a header field to the headers object</p>
200  * 
201  * @param {String} key Key name
202  * @param {String} value Header value
203  */
204 MailComposer.prototype.addHeader = function(key, value){
205     key = this.normalizeKey(key);
206     value = (value ||Â "").toString().trim();
207     if(!key || !value){
208         return;
209     }
210     
211     if(!(key in this._headers)){
212         this._headers[key] = value;
213     }else{
214         if(!Array.isArray(this._headers[key])){
215             this._headers[key] = [this._headers[key], value];
216         }else{
217             this._headers[key].push(value);
218         }
219     }
220 };

222 /**
223  * <p>Gets a header field</p>
224  * 
225  * @param {String} key Key name
226  * @return {String|Array} Header field - if several values, then it's an array
227  */
228 MailComposer.prototype.getHeader = function(key){
229     var value;
230     
231     key = this.normalizeKey(key);
232     value = this._headers[key] || "";
233     
234     return value;
235 };

237 /**
238  * <p>Adds an attachment to the list</p>
239  * 
240  * <p>Following options are allowed:</p>
241  * 
242  * <ul>
243  *     <li><b>fileName</b> - filename for the attachment</li>
244  *     <li><b>contentType</b> - content type for the attachmetn (default will be derived from the filename)</li>
245  *     <li><b>cid</b> - Content ID value for inline images</li>
246  *     <li><b>contents</b> - String or Buffer attachment contents</li>
247  *     <li><b>filePath</b> - Path to a file for streaming</li>
248  * </ul>
249  * 
250  * <p>One of <code>contents</code> or <code>filePath</code> must be specified, otherwise
251  * the attachment is not included</p>
252  * 
253  * @param {Object} attachment Attachment info
254  */
255 MailComposer.prototype.addAttachment = function(attachment){
256     attachment = attachment ||Â {};
257     var filename;
258     
259     // Needed for Nodemailer compatibility
260     if(attachment.filename){
261         attachment.fileName = attachment.filename;
262         delete attachment.filename;
263     }
264     
265     if(!attachment.contentType){
266         filename = attachment.fileNameÂ ||Â attachment.filePath;
267         if(filename){
268             attachment.contentType = this.getMimeType(filename);
269         }else{
270             attachment.contentType = "application/octet-stream";
271         }
272     }
273     
274     if(attachment.filePath || attachment.contents){
275         this._attachments.push(attachment);
276     }
277 };

291 /**
292  * <p>Composes and emits an envelope from the <code>this._envelope</code> 
293  * object. Needed for the SMTP client</p>
294  * 
295  * <p>Emitted envelope is int hte following structure:</p>
296  * 
297  * <pre>
298  * {
299  *     to: "address",
300  *     from: ["list", "of", "addresses"]
301  * }
302  * </pre>
303  * 
304  * <p>Both properties (<code>from</code> and <code>to</code>) are optional
305  * and may not exist</p>
306  */
307 MailComposer.prototype.composeEnvelope = function(){
308     var envelope = {},
309         toKeys = ["to", "cc", "bcc"],
310         key;
311     
312     // If multiple addresses, only use the first one
313     if(this._envelope.from && this._envelope.from.length){
314         envelope.from = [].concat(this._envelope.from).shift();
315     }
316     
317     for(var i=0, len=toKeys.length; i<len; i++){
318         key = toKeys[i];
319         if(this._envelope[key] && this._envelope[key].length){
320             if(!envelope.to){
321                 envelope.to = [];
322             }
323             envelope.to = envelope.to.concat(this._envelope[key]);
324         }
325     }
326     
327     this.emit("envelope", envelope);
328 };

330 /**
331  * <p>Composes a header for the message and emits it with a <code>'data'</code>
332  * event</p>
333  * 
334  * <p>Also checks and build a structure for the message (is it a multipart message
335  * and does it need a boundary etc.)</p>
336  * 
337  * <p>By default the message is not a multipart. If the message containes both
338  * plaintext and html contents, an alternative block is used. it it containes
339  * attachments, a mixed block is used. If both alternative and mixed exist, then
340  * alternative resides inside mixed.</p>
341  */
342 MailComposer.prototype.composeHeader = function(){
343     var keys, key,
344         contentType,
345         headers = [];
346     
347     
348     
349     if(this._message.body && this._message.html){
350         this._message.useAlternative = true;
351         this._message.alternativeBoundary = "----NODEMAILER-?=_"+(++this._gencounter)+"-"+Date.now();
352     }else{
353         this._message.useAlternative = false;
354     }
355     
356     if(this._attachments.length){
357         this._message.useMixed = true;
358         this._message.mixedBoundary = "----NODEMAILER-?=_"+(++this._gencounter)+"-"+Date.now();
359     }else{
360         this._message.useMixed = false;
361     }
362     
363     if(this._message.useMixed){
364         this._messageType = "multipart/mixed";
365         this._messageBoundary = this._message.mixedBoundary;
366     }else if(this._message.useAlternative){
367         this._messageType = "multipart/alternative";
368         this._messageBoundary = this._message.alternativeBoundary;
369     }else if(this._message.html){
370         this._messageType = "text/html; charset=utf-8";
371         this._message.encoding = this.options.encoding || "quoted-printable";
372     }else{
373         this._messageType = "text/plain; charset=utf-8";
374         this._message.encoding = this.options.encoding || "quoted-printable";
375     }
376     
377     if(this._messageBoundary){
378         contentType = this._messageType+"; boundary=\""+this._messageBoundary+"\"";
379     }else{
380         contentType = this._messageType;
381     }
382     
383     this.addHeader("Content-Type", this._messageType);
384     this.addHeader("Content-Transfer-Encoding", this._message.encoding);
385     
386     // FROM
387     if(this._message.from && this._message.from.length){
388         [].concat(this._message.from).forEach((function(from){
389             this.addHeader("From", from);
390         }).bind(this));
391     }
392     
393     // TO
394     if(this._message.to && this._message.to.length){
395         [].concat(this._message.to).forEach((function(to){
396             this.addHeader("To", to);
397         }).bind(this));
398     }
399     
400     // CC
401     if(this._message.cc && this._message.cc.length){
402         [].concat(this._message.cc).forEach((function(cc){
403             this.addHeader("Cc", cc);
404         }).bind(this));
405     }
406     
407     // REPLY-TO
408     if(this._message.replyTo && this._message.replyTo.length){
409         [].concat(this._message.replyTo).forEach((function(replyTo){
410             this.addHeader("Reply-To", replyTo);
411         }).bind(this));
412     }
413     
414     // SUBJECT
415     if(this._message.subject){
416         this.addHeader("Subject", this._message.subject);
417     }
418     
419     // Compile header lines
420     keys = Object.keys(this._headers);
421     for(var i=0, len = keys.length; i<len; i++){
422         key = this.normalizeKey(keys[i]);
423         
424         headers = headers.concat([].concat(this._headers[key]).map(function(value){
425             return key+": "+value;
426         }));
427     }
428     
429     this.emit("data", new Buffer(headers.join("\r\n")+"\r\n\r\n", "utf-8"));
430 };

432 /*
433  * <p>Normalizes a key nam by cpitalizing first chars of words</p>
434  * 
435  * <p><code>x-mailer</code> will become <code>X-Mailer</code></p>
436  * 
437  * <p>Needed to avoid duplicate header keys</p>
438  * 
439  * @param {String} key Key name
440  * @return {String} First chars uppercased
441  */
442 MailComposer.prototype.normalizeKey = function(key){
443     return (key || "").toString().trim().
444         toLowerCase().
445         replace(/^\S|[\-\s]\S/g, function(c){
446             return c.toUpperCase();
447         }).replace(/^MIME\-/i, "MIME-");
448 };

450 /**
451  * <p>Tests if a string has high bit (UTF-8) symbols</p>
452  * 
453  * @param {String} str String to be tested for high bit symbols
454  * @return {Boolean} true if high bit symbols were found
455  */
456 MailComposer.prototype.hasUTFChars = function(str){
457     var rforeign = /[^\u0000-\u007f]/;
458     return !!rforeign.test(str);
459 };

461 /**
462  * <p>Converts a string to mime word format. If the length is longer than
463  * <code>maxlen</code>, split it</p>
464  * 
465  * <p>If the string doesn't have any unicode characters return the original 
466  * string instead</p>
467  * 
468  * @param {String} str String to be encoded
469  * @param {String} encoding Either Q for Quoted-Printable or B for Base64
470  * @param {Number} [maxlen] Optional length of the resulting string, whitespace will be inserted if needed
471  * 
472  * @return {String} Mime-word encoded string (if needed)
473  */
474 MailComposer.prototype.encodeMimeWord = function(str, encoding, maxlen){
475     encoding = (encoding || "Q").toUpperCase(); 
476     if(this.hasUTFChars(str)){
477         str = mimelib.encodeMimeWord(str, encoding);
478         if(maxlen && str.length>maxlen){
479             return str.replace(new RegExp(".{"+maxlen+"}","g"),"$&?= =?UTF-8?"+encoding+"?");
480         }else{
481             return str;
482         }
483     }else{
484         return str;
485     }
486 };

488 /**
489  * <p>Resolves a mime type for a filename</p>
490  * 
491  * @param {String} filename Filename to check
492  * @return {String} Corresponding mime type
493  */
494 MailComposer.prototype.getMimeType = function(filename){
495     var defaultMime = "application/octet-stream",
496         extension = filename && filename.substr(filename.lastIndexOf(".")+1).trim().toLowerCase();
497     return extension && mimelib.contentTypes[extension] || defaultMime;
